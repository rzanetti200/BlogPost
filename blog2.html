.typewriter {
        font-family: Courier, monospace;
        display: inline-block;
    }
    
    .typewriter-text {
        display: inline-block;
          overflow: hidden;
          letter-spacing: 2px;
         animation: typing 5s steps(30, end), blink .75s step-end infinite;
        white-space: nowrap;
        font-size: 30px;
        font-weight: 700;
        border-right: 4px solid red;
        box-sizing: border-box;
    }
    
    @keyframes typing {
        from { 
            width: 0% 
        }
        to { 
            width: 100% 
        }
    }
    
    @keyframes blink {
        from, to { 
            border-color: transparent 
        }
        50% { 
            border-color: red; 
        }
    }
